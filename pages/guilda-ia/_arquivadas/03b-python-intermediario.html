Estruturas de Dados Avançadas

Comprehensions

Forma concisa de criar listas, dicionários e sets. 

# List comprehension
numeros = [1, 2, 3, 4, 5]
quadrados = [n ** 2 for n in numeros]  # [1, 4, 9, 16, 25]

# Com condição
pares = [n for n in numeros if n % 2 == 0]  # [2, 4]

# Dict comprehension
palavras = ["gato", "cachorro", "pássaro"]
tamanhos = {p: len(p) for p in palavras}  # {"gato": 4, "cachorro": 8, ...}

Generators

Funções que produzem valores sob demanda, economizando memória. 

# Generator function
def contar_ate(n):
    for i in range(n):
        yield i

# Uso
for num in contar_ate(5):
    print(num)  # 0, 1, 2, 3, 4

# Generator expression
quadrados_gen = (n ** 2 for n in range(1000000))  # Não cria lista na memória

Quando usar Generator vs Lista

Situação Use Lista Use Generator
Precisa acessar múltiplas vezes ✓ ✗
Precisa de len() ou indexação ✓ ✗
Grande volume de dados ✗ ✓
Pipeline de processamento ✗ ✓

Classes Básicas

class Agente:
    """Representa um agente de IA com memória."""

    def __init__(self, nome: str, sistema: str = "Você é um assistente."):
        """
        Inicializa o agente.

        Args:
            nome: Nome identificador do agente
            sistema: System prompt do agente
        """
        self.nome = nome
        self.sistema = sistema
        self.historico = []

    def conversar(self, mensagem: str) -> str:
        """Processa uma mensagem e retorna resposta."""
        self.historico.append({"role": "user", "content": mensagem})
        # ... lógica de processamento ...
        resposta = f"[{self.nome}] Resposta simulada"
        self.historico.append({"role": "assistant", "content": resposta})
        return resposta

    def limpar_memoria(self) -> None:
        """Limpa o histórico de conversa."""
        self.historico = []
        return "Memória limpa"

    def __repr__(self) -> str:
        return f"Agente(nome='{self.nome}', mensagens={len(self.historico)})"


# Uso
agente = Agente("Assistente", "Você é um tutor de Python.")
print(agente.conversar("Olá!"))  # [Assistente] Resposta simulada
print(agente)  # Agente(nome='Assistente', mensagens=2)

Herança

class AgenteComFerramentas(Agente):
    """Agente que pode usar ferramentas externas."""

    def __init__(self, nome: str, sistema: str = "Você é um assistente.", ferramentas: dict = None):
        super().__init__(nome, sistema)
        self.ferramentas = ferramentas or {}

    def adicionar_ferramenta(self, nome: str, funcao: callable) -> None:
        """Registra uma nova ferramenta."""
        self.ferramentas[nome] = funcao

    def usar_ferramenta(self, nome: str, *args, **kwargs):
        """Executa uma ferramenta pelo nome."""
        if nome not in self.ferramentas:
            raise ValueError(f"Ferramenta '{nome}' não encontrada")
        return self.ferramentas[nome](*args, **kwargs)


# Uso
def calculadora(expr: str):
    return eval(expr)  # Simplificado - não usar em produção!

agente = AgenteComFerramentas("Calc", ferramentas={"calc": calculadora})
print(agente.usar_ferramenta("calc", "2 + 2"))  # 4

@property e Encapsulamento

class AgenteProtegido:
    def __init__(self, nome: str):
        self._nome = nome  # "Privado" por convenção
        self.__historico = []  # Name-mangled (mais privado)

    @property
    def nome(self) -> str:
        """Getter para nome."""
        return self._nome

    @nome.setter
    def nome(self, valor: str) -> None:
        """Setter com validação."""
        if not valor.strip():
            raise ValueError("Nome não pode ser vazio")
        self._nome = valor

    @property
    def total_mensagens(self) -> int:
        """Propriedade calculada."""
        return len(self.__historico)

Context Managers

Padrão para gerenciar recursos automaticamente. 

from contextlib import contextmanager

@contextmanager
def agente_contexto(nome: str):
    """Context manager para agente com cleanup automático."""
    agente = Agente(nome)
    print(f"[{nome}] Iniciando sessão...")
    try:
        yield agente
    finally:
        print(f"[{nome}] Encerrando sessão...")
        agente.limpar_memoria()

# Uso
with agente_contexto("TestBot") as agente:
    agente.conversar("Olá!")
    # Automaticamente faz cleanup ao sair

Padrões Úteis para Agentes

Singleton para Memória Compartilhada

class MemoriaGlobal:
    _instancia = None

    def __new__(cls):
        if cls._instancia is None:
            cls._instancia = super().__new__(cls)
            cls._instancia._dados = {}
        return cls._instancia

    def salvar(self, chave: str, valor: any) -> None:
        self._dados[chave] = valor

    def recuperar(self, chave: str) -> any:
        return self._dados.get(chave)

# Uso
mem1 = MemoriaGlobal()
mem2 = MemoriaGlobal()
mem1.salvar("teste", 123)
print(mem2.recuperar("teste"))  # 123 - mesma instância!

class AgenteFactory:

        return AgenteComFerramentas(
            nome="Assistente",
            sistema="Você é um assistente útil e educado.",
            ferramentas={"calc": calculadora, "horario": obter_horario}
        )

    @staticmethod
    def criar_tutor_python() -> AgenteComFerramentas:
        return AgenteComFerramentas(
            nome="TutorPython",
            sistema="Você é um professor de Python paciente e didático.",
            ferramentas={"calc": calculadora}
        )

    @staticmethod
    def criar_do_tipo(tipo: str) -> AgenteComFerramentas:
        """Cria agente baseado em string."""
        criadores = {
            "assistente": AgenteFactory.criar_assistente,
            "tutor_python": AgenteFactory.criar_tutor_python,
        }
        if tipo not in criadores:
            raise ValueError(f"Tipo de agente desconhecido: {tipo}")
        return criadores[tipo]()

# Uso
agente = AgenteFactory.criar_do_tipo("tutor_python")

from dataclasses import dataclass

@dataclass
class Config:
    nome: str
    sistema: str
    max_tokens: int = 1000
    temperatura: float = 0.7

class AgenteConfiguravel:
    def __init__(self, config: Config):
        if not config.nome.strip():
            raise ValueError("Nome não pode ser vazio")
        self.config = config
        self.historico: list = []

    def conversar(self, mensagem: str) -> str:
        self.historico.append({"role": "user", "content": mensagem})
        resposta = f"[{self.config.nome}] Processando com temp={self.config.temperatura}"
        self.historico.append({"role": "assistant", "content": resposta})
        return resposta

    @property
    def total_mensagens(self) -> int:
        return len(self.historico)

Exercício 2: Repository Pattern para Histórico

Implemente um RepositorioHistorico que salva/recupera mensagens:

📝 Gabarito 

import json
from pathlib import Path
from typing import List, Dict

class RepositorioHistorico:
    def __init__(self, arquivo: str = "historico.json"):
        self.arquivo = Path(arquivo)

    def salvar(self, mensagens: List[Dict[str, str]]) -> None:
        with open(self.arquivo, 'w', encoding='utf-8') as f:
            json.dump(mensagens, f, ensure_ascii=False, indent=2)

    def carregar(self) -> List[Dict[str, str]]:
        if not self.arquivo.exists():
            return []
        with open(self.arquivo, 'r', encoding='utf-8') as f:
            return json.load(f)

    def limpar(self) -> None:
        if self.arquivo.exists():
            self.arquivo.unlink()

# Uso
repo = RepositorioHistorico()
repo.salvar([{"role": "user", "content": "Olá"}])
mensagens = repo.carregar()
print(mensagens)
